# Fonction de remplissage (padding)
def remplissage(tableau):
    S = tableau.shape[0]
    R = tableau.shape[1]
    new_tableau = tableau.copy()
    while S % 8 != 0:
        new_tableau = np.vstack((new_tableau, np.zeros((1, R, 3), dtype=new_tableau.dtype)))
        S += 1
    while R % 8 != 0:
        new_tableau = np.hstack((new_tableau, np.zeros((S, 1, 3), dtype=new_tableau.dtype)))
        R += 1
    return new_tableau

# Fonction de décomposition en matrices Y, Cb, Cr
def decomposition(tableau):
    hauteur = tableau.shape[0]
    largeur = tableau.shape[1]
    Y = np.zeros((hauteur, largeur), dtype=tableau.dtype)
    Cb = np.zeros((hauteur, largeur), dtype=tableau.dtype)
    Cr = np.zeros((hauteur, largeur), dtype=tableau.dtype)
    for i in range(hauteur):
        for j in range(largeur):
            Y[i, j] = tableau[i][j][0]
            Cb[i, j] = tableau[i][j][1]
            Cr[i, j] = tableau[i][j][2]
    return Y, Cb, Cr

# Fonction de découpage en blocs 8x8
def decoupage_block(tableau):
    l = []
    for i in range(0, tableau.shape[0], 8):
        for j in range(0, tableau.shape[1], 8):
            block = tableau[i:i+8, j:j+8]
            l.append(block)
    return l

# Fonction de centrage des blocs (soustraction de 128)
def centre(block):
    for i in range(8):
        for j in range(8):
            block[i][j] = int(block[i][j]) - 128
    return block

# Fonction d'échantillonnage (sous-échantillonnage 4:2:0)
def echantillonage(block):
    for i in range(4):
        for j in range(4):
            S = (block[i*2][j*2] + block[i*2+1][j*2] + block[i*2][j*2+1] + block[i*2+1][j*2+1]) / 4
            block[i*2][j*2] = S
            block[i*2+1][j*2] = S
            block[i*2][j*2+1] = S
            block[i*2+1][j*2+1] = S
    return block

# Fonction décomposition corrigée, conservant la structure avec deux boucles for imbriquées
def decomposition(tableau):
    Y = np.array([])
    cb = np.array([])
    cr = np.array([])
    
    for i in range(tableau.shape[0]):
        n1 = np.array([])  # valeurs Y de la ligne i
        n2 = np.array([])  # valeurs Cb de la ligne i
        n3 = np.array([])  # valeurs Cr de la ligne i
        
        for j in range(tableau.shape[1]):
            n1 = np.append(n1, tableau[i][j][0])
            n2 = np.append(n2, tableau[i][j][1])
            n3 = np.append(n3, tableau[i][j][2])
        
        Y = np.append(Y, n1)
        cb = np.append(cb, n2)
        cr = np.append(cr, n3)
    
    # Remettre les tableaux dans la forme 2D (hauteur x largeur)
    hauteur, largeur = tableau.shape[0], tableau.shape[1]
    Y = Y.reshape((hauteur, largeur))
    cb = cb.reshape((hauteur, largeur))
    cr = cr.reshape((hauteur, largeur))
    
    return Y, cb, cr


# Fonction DCT (Direct Cosine Transform) sur un block 8x8, formule brute
def conversion_DCT(block):
    B = np.zeros((8, 8))
    for i in range(8):
        for j in range(8):
            S = 0
            for k in range(8):
                for e in range(8):
                    cosA = m.cos(((2 * k + 1) * np.pi * i) / 16)
                    cosB = m.cos(((2 * e + 1) * np.pi * j) / 16)
                    S += block[k][e] * cosA * cosB
            a = 1 / m.sqrt(2) if i == 0 else 1
            b = 1 / m.sqrt(2) if j == 0 else 1
            B[i][j] = 0.25 * a * b * S  # (1/4) = 1/sqrt(8*8)
    return B

# Fonction DCT avec scipy
def dct_sci(block):
    return dct(dct(block.T, norm='ortho').T, norm='ortho')

# Fonction de quantification
def quantification(block, quant_matrix):
    return np.round(block / quant_matrix).astype(int)

# Fonction Zig-Zag
def zig_zag(block):
    l = []
    indices = [
        (0,0),
        (0,1),(1,0),
        (2,0),(1,1),(0,2),
        (0,3),(1,2),(2,1),(3,0),
        (4,0),(3,1),(2,2),(1,3),(0,4),
        (0,5),(1,4),(2,3),(3,2),(4,1),(5,0),
        (6,0),(5,1),(4,2),(3,3),(2,4),(1,5),(0,6),
        (0,7),(1,6),(2,5),(3,4),(4,3),(5,2),(6,1),(7,0),
        (7,1),(6,2),(5,3),(4,4),(3,5),(2,6),(1,7),
        (2,7),(3,6),(4,5),(5,4),(6,3),(7,2),
        (7,3),(6,4),(5,5),(4,6),(3,7),
        (4,7),(5,6),(6,5),(7,4),
        (7,5),(6,6),(5,7),
        (6,7),(7,6),
        (7,7)
    ]
    for (i,j) in indices:
        l.append(block[i][j])
    return l

# Fonction RLE sans tri préalable
def RLE(liste):
    l1 = []
    p = 0
    while p < len(liste):
        i = p + 1
        while i < len(liste) and liste[i] == liste[i - 1]:
            i += 1
        l1.append((liste[p], i - p))
        p = i
    return l1

# Fonction tri décroissant sur le deuxième élément d'une liste de tuples
def tri_par_deuxieme_decroissant(liste):
    return sorted(liste, key=lambda x: x[1], reverse=True)

# Calcul des fréquences pour Huffman à partir de la liste RLE (on compte juste la fréquence des symboles)
def frequences_rle(rle_liste):
    freq_dict = {}
    for symbole, count in rle_liste:
        freq_dict[symbole] = freq_dict.get(symbole, 0) + 1
    return list(freq_dict.items())

# Construction de l’arbre de Huffman
def construire_arbre_huffman(freqs):
    heap = [[freq, [symbole, ""]] for symbole, freq in freqs]
    heapq.heapify(heap)
    while len(heap) > 1:
        lo = heapq.heappop(heap)
        hi = heapq.heappop(heap)
        for paire in lo[1:]:
            paire[1] = '0' + paire[1]
        for paire in hi[1:]:
            paire[1] = '1' + paire[1]
        heapq.heappush(heap, [lo[0] + hi[0]] + lo[1:] + hi[1:])
    return {symbole: code for symbole, code in heap[0][1:]}

# Fonction Huffman complète depuis les données brutes (sans tri préalable)
def huffman_depuis_rle(texte):
    rle_liste = RLE(texte)  # RLE sur la liste originale, sans tri
    freqs = frequences_rle(rle_liste)
    freqs = tri_par_deuxieme_decroissant(freqs)
    code_map = construire_arbre_huffman(freqs)
    
    encodage = ""
    for symbole, count in rle_liste:
        encodage += code_map[symbole] * count  # on encode le symbole répété count fois
    
    return code_map, encodage

# Déjà donné : inverse zig-zag
def inverse_zig_zag(l):
    block = np.zeros((8, 8))
    order = [
        (0, 0), (1, 0), (0, 1), (0, 2), (1, 1), (2, 0), (3, 0), (2, 1),
        (1, 2), (0, 3), (0, 4), (1, 3), (2, 2), (3, 1), (4, 0), (5, 0),
        (4, 1), (3, 2), (2, 3), (1, 4), (0, 5), (0, 6), (1, 5), (2, 4),
        (3, 3), (4, 2), (5, 1), (6, 0), (7, 0), (6, 1), (5, 2), (4, 3),
        (3, 4), (2, 5), (1, 6), (0, 7), (1, 7), (2, 6), (3, 5), (4, 4),
        (5, 3), (6, 2), (7, 1), (7, 2), (6, 3), (5, 4), (4, 5), (3, 6),
        (2, 7), (3, 7), (4, 6), (5, 5), (6, 4), (7, 3), (7, 4), (6, 5),
        (5, 6), (4, 7), (5, 7), (6, 6), (7, 5), (7, 6), (7, 7)
    ]
    for idx, pos in enumerate(order):
        block[pos] = l[idx]
    return block

# Déjà donné : inverse DCT
def idct_sci(block):
    return idct(idct(block.T, norm='ortho').T, norm='ortho')

# RLE inverse (reconstruction de la liste à partir des tuples (valeur, count))
def RLE_inverse(rle_liste):
    resultat = []
    for val, count in rle_liste:
        resultat.extend([val] * count)
    return resultat

# Fonction de décodage Huffman
def decodage_huffman(encodage, table_codes):
    # Inverse la table {symbole: code} -> {code: symbole}
    table_inverse = {code: symbole for symbole, code in table_codes.items()}
    
    resultat = []
    tampon = ''
    
    for bit in encodage:
        tampon += bit
        if tampon in table_inverse:
            resultat.append(table_inverse[tampon])
            tampon = ''
    return resultat

# Fonction déquantification (inverse de quantification)
def dequantification(block, quant_matrix):
    return block * quant_matrix

# Fonction de recentrage (ajouter 128)
def recentrage(block):
    return block + 128

# Fonction pour reconstituer les blocs 8x8 dans une image 2D
def reassembler_image(blocs, hauteur, largeur):
    image = np.zeros((hauteur, largeur))
    blocs_par_ligne = largeur // 8
    k = 0
    for ligne_bloc in range(hauteur // 8):
        for colonne_bloc in range(blocs_par_ligne):
            bloc = blocs[k]
            start_i = ligne_bloc * 8
            start_j = colonne_bloc * 8
            image[start_i:start_i+8, start_j:start_j+8] = bloc
            k += 1
    return image

            bloc_zig = flat_coeffs[i:i+64]
